package com.pholser.junit.quickcheck.generator.java.nio.charset;

import static java.nio.charset.Charset.availableCharsets;

import com.pholser.junit.quickcheck.generator.GenerationStatus;
import com.pholser.junit.quickcheck.generator.Generator;
import com.pholser.junit.quickcheck.random.SourceOfRandomness;
import java.nio.charset.Charset;

/**
 * Produces values of type {@link java.nio.charset.Charset}.
 */
public class CharsetGenerator extends Generator<Charset> {
    public CharsetGenerator() {
        super(Charset.class);
    }

    @Override public Charset generate(
        SourceOfRandomness random,
        GenerationStatus status) {

        return Charset.forName(
            random.choose(availableCharsets().keySet()));
    }
}
